
CREATE TABLE `user` (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    coin_balance INT DEFAULT 0,
    author_balance INT DEFAULT 0
);

CREATE TABLE genres (
    genre_id INT AUTO_INCREMENT PRIMARY KEY,
    genre_name VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE story (
    story_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    title VARCHAR(150) NOT NULL,
    synopsis TEXT,
    genre_id INT,
    tags VARCHAR(255),
    status ENUM('draft', 'published') DEFAULT 'draft',
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
    FOREIGN KEY (genre_id) REFERENCES genres(genre_id) ON DELETE SET NULL
);

CREATE TABLE chapter (
    chapter_id INT AUTO_INCREMENT PRIMARY KEY,
    story_id INT,
    chapter_number INT NOT NULL,
    chapter_title VARCHAR(150) NOT NULL,
    is_premium BOOLEAN DEFAULT FALSE,
    coin_cost INT DEFAULT 0,
    status ENUM('draft', 'published') DEFAULT 'draft',
    FOREIGN KEY (story_id) REFERENCES story(story_id) ON DELETE CASCADE
);

CREATE TABLE `character` (
    character_id INT AUTO_INCREMENT PRIMARY KEY,
    story_id INT,
    character_name VARCHAR(100) NOT NULL,
    FOREIGN KEY (story_id) REFERENCES story(story_id) ON DELETE CASCADE
);

CREATE TABLE purchase_chapter (
    purchase_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    chapter_id INT,
    coins_spent INT NOT NULL,
    purchase_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
    FOREIGN KEY (chapter_id) REFERENCES chapter(chapter_id) ON DELETE CASCADE
);

CREATE TABLE transaction_topup (
    topup_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    amount_paid INT NOT NULL,
    coins_gained INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE
);

CREATE TABLE withdrawal (
    withdrawal_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    amount INT NOT NULL,
    status ENUM('pending', 'approved', 'rejected') DEFAULT 'pending',
    requested_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE
);

CREATE TABLE transaction_history (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    type ENUM('purchase', 'withdrawal', 'top_up') NOT NULL,
    amount DECIMAL(10,2) NOT NULL,
    description VARCHAR(255) DEFAULT NULL,
    reference_id INT DEFAULT NULL,
    reference_table VARCHAR(50) DEFAULT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES user(user_id) ON DELETE CASCADE
);

-- INTERAKSI PEMBACA 
CREATE TABLE read_history (
    history_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    chapter_id INT,
    read_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
    FOREIGN KEY (chapter_id) REFERENCES chapter(chapter_id) ON DELETE CASCADE
);

CREATE TABLE bookmark (
    bookmark_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    story_id INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
    FOREIGN KEY (story_id) REFERENCES story(story_id) ON DELETE CASCADE
);

CREATE TABLE library (
    library_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    story_id INT,
    last_read_chapter_id INT NULL,
    added_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
    FOREIGN KEY (story_id) REFERENCES story(story_id) ON DELETE CASCADE,
    FOREIGN KEY (last_read_chapter_id) REFERENCES chapter(chapter_id) ON DELETE SET NULL
);

CREATE TABLE votes (
    vote_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    chapter_id INT,
    voted_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY unique_vote (user_id, chapter_id),
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
    FOREIGN KEY (chapter_id) REFERENCES chapter(chapter_id) ON DELETE CASCADE
);

CREATE TABLE subscriptions (
    sub_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    expiry_date DATETIME NOT NULL,
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE
);

CREATE TABLE contest_submissions (
    submission_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    story_id INT,
    status ENUM('pending', 'approved', 'rejected') DEFAULT 'pending',
    submitted_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
    FOREIGN KEY (story_id) REFERENCES story(story_id) ON DELETE CASCADE
);

-- TRIGGER 
DELIMITER $

-- 1. TRIGGER SAAT PEMBELIAN CHAPTER (tetap, sama)
CREATE TRIGGER after_purchase_chapter_insert
AFTER INSERT ON purchase_chapter
FOR EACH ROW
BEGIN
    UPDATE `user`
    SET coin_balance = coin_balance - NEW.coins_spent
    WHERE user_id = NEW.user_id;

    UPDATE `user`
    SET author_balance = author_balance + NEW.coins_spent
    WHERE user_id = (
        SELECT s.user_id FROM story s JOIN chapter c ON s.story_id = c.story_id WHERE c.chapter_id = NEW.chapter_id
    );

    -- Catat ke transaction_history
    INSERT INTO transaction_history (user_id, type, amount, description, reference_id, reference_table)
    VALUES (NEW.user_id, 'purchase', NEW.coins_spent, CONCAT('Buy chapter #', NEW.chapter_id), NEW.purchase_id, 'purchase_chapter');
END $

-- 2. TRIGGER SAAT TOP-UP (otomatis catat history)
CREATE TRIGGER after_topup_insert
AFTER INSERT ON transaction_topup
FOR EACH ROW
BEGIN
    INSERT INTO transaction_history (user_id, type, amount, description, reference_id, reference_table)
    VALUES (NEW.user_id, 'top_up', NEW.coins_gained, CONCAT('Top-up Rp', NEW.amount_paid), NEW.topup_id, 'transaction_topup');
END $

DELIMITER ;

-- STORED PROCEDURE 
DELIMITER $

-- 3. APPROVE WITHDRAWAL (validasi + potong saldo + catat history)
CREATE PROCEDURE approve_withdrawal(IN p_withdrawal_id INT)
BEGIN
    DECLARE v_user_id INT;
    DECLARE v_amount INT;
    DECLARE v_status VARCHAR(20);
    DECLARE v_balance DECIMAL(10,2);

    -- Ambil data withdrawal
    SELECT user_id, amount, status INTO v_user_id, v_amount, v_status
    FROM withdrawal WHERE withdrawal_id = p_withdrawal_id;

    IF v_user_id IS NULL THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Withdrawal tidak ditemukan';
    END IF;

    IF v_status != 'pending' THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Withdrawal sudah diproses sebelumnya';
    END IF;

    -- Cek saldo author
    SELECT author_balance INTO v_balance FROM user WHERE user_id = v_user_id;
    IF v_balance < v_amount THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Saldo author tidak mencukupi';
    END IF;

    -- Update status & potong saldo
    UPDATE withdrawal SET status = 'approved' WHERE withdrawal_id = p_withdrawal_id;
    UPDATE user SET author_balance = author_balance - v_amount WHERE user_id = v_user_id;

    -- Catat history
    INSERT INTO transaction_history (user_id, type, amount, description, reference_id, reference_table)
    VALUES (v_user_id, 'withdrawal', v_amount, CONCAT('Withdrawal approved'), p_withdrawal_id, 'withdrawal');
END $

-- 4. REJECT WITHDRAWAL (tidak potong saldo, tetap catat history)
CREATE PROCEDURE reject_withdrawal(IN p_withdrawal_id INT)
BEGIN
    DECLARE v_status VARCHAR(20);

    SELECT status INTO v_status FROM withdrawal WHERE withdrawal_id = p_withdrawal_id;

    IF v_status IS NULL THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Withdrawal tidak ditemukan';
    END IF;

    IF v_status != 'pending' THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Withdrawal sudah diproses sebelumnya';
    END IF;

    UPDATE withdrawal SET status = 'rejected' WHERE withdrawal_id = p_withdrawal_id;

    INSERT INTO transaction_history (user_id, type, amount, description, reference_id, reference_table)
    VALUES ((SELECT user_id FROM withdrawal WHERE withdrawal_id = p_withdrawal_id), 'withdrawal', 0, CONCAT('Withdrawal rejected'), p_withdrawal_id, 'withdrawal');
END $

DELIMITER ;

-- VIEW 
CREATE VIEW author_dashboard AS
SELECT
    s.user_id AS author_id,
    s.story_id,
    s.title AS story_title,
    COUNT(DISTINCT c.chapter_id) AS total_chapters,
    COUNT(pc.purchase_id) AS total_chapters_sold,
    COALESCE(SUM(pc.coins_spent), 0) AS total_revenue_coins
FROM story s
LEFT JOIN chapter c ON s.story_id = c.story_id
LEFT JOIN purchase_chapter pc ON c.chapter_id = pc.chapter_id
GROUP BY s.story_id;
